```js
Function.prototype.myCall = function () {
    let ctx = [...arguments].shift() || window;
    ctx.fn = this;
    let arg = [...arguments].slice(1);
    let result = ctx.fn(...arg);
    delete ctx.fn
    return result

}
let obj = {x: 1};
function tst(a) {console.log(this.x, a)};
```
